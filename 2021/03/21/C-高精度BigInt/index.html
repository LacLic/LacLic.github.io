C++高精度BigInt

Posted on 2021-03-21 Edited on 2021-03-22

代码

#include <cstdio>
#include <cstring>
#define TYPE char // type of BigInt element
#define LENGTH 10100 // N==50000, LENGTH == 9986
const int bigger = -1;
const int equal = 0;
const int smaller = 1;
const int base = 10;

class BigInt {
    private:
    TYPE n[LENGTH];
    int bit;
    bool isPositive = true;
    BigInt big_minus_small(const BigInt&o1,const BigInt&o2,const int&maxbit,const int&minbit,int bias=0) {
        BigInt ret;
        for(int i=0;i<bias;++i) {
            ret.n[i] = o1.n[i];
        }
        int down = 0;
        for(int i=0;i<minbit;++i) {
            ret.n[i+bias] = o1.n[i+bias] - o2.n[i] - down;
            if(ret.n[i+bias]<0) {
                down = 1;
                ret.n[i+bias] += base;
            }else {
                down = 0;
            }
        }
        for(int i=minbit;i<maxbit;++i) {
            ret.n[i+bias] = o1.n[i+bias] - down;
            if(ret.n[i+bias]<0) {
                down = 1;
                ret.n[i+bias] += base;
            }else {
                down = 0;
            }
        }
        ret.bit=1;
        for(int i=maxbit-1;i>=0;--i) {
            if(ret.n[i]!=0) {
                ret.bit = i+1;
                break;
            }
        }
        return ret;
    }

    bool biasSmaller(const BigInt&o2, int bias) {
        if(this->bit<o2.bit+bias) {
            return true;
        }else if(this->bit>o2.bit+bias) {
            return false;
        }else {
            for(int i=o2.bit-1;i>=0;--i) {
                if(this->n[i+bias] < o2.n[i]) {
                    return true;
                }else if(this->n[i+bias] > o2.n[i]) {
                    return false;
                }
            }
            return false; // equal
        }
    }

    int compare(const BigInt&o1, const BigInt&o2) {
        if(this->bit<o2.bit) {
            return smaller;
        }else if(this->bit>o2.bit) {
            return bigger;
        }else {
            for(int i=o2.bit-1;i>=0;--i) {
                if(this->n[i] < o2.n[i]) {
                    return smaller;
                }else if(this->n[i] > o2.n[i]) {
                    return bigger;
                }
            }
            return equal;
        }
    }

    public:
    BigInt() {}

    BigInt(unsigned long long val) {
        int cnt=0;
        do {
            this->n[cnt] = val;
            val/=base;
            ++cnt;
        }while(val);
        this->bit = cnt;
    }

    bool operator<(const BigInt&o2) {
        if(compare(*this,o2)==smaller) {
            return true;
        }else {
            return false;
        }
    }

    bool operator>=(const BigInt&o2) {
        return !(*this<o2);
    }

    bool operator>(const BigInt&o2) {
        if(compare(*this,o2)==bigger) {
            return true;
        }else {
            return false;
        }
    }

    bool operator<=(const BigInt&o2) {
        return !(*this>o2);
    }

    bool operator==(const BigInt&o2) {
        if(this->bit!=o2.bit) {
            return false;
        }else {
            for(int i=this->bit-1;i>=0;++i) {
                if(this->n[i]!=o2.n[i]) {
                    return false;
                }
            }
            return true;
        }
    }

    BigInt operator+(const BigInt&o2) {
        BigInt ret;
        int maxbit,minbit,isThis;
        if(this->bit > o2.bit) {
            maxbit = this->bit;
            minbit = o2.bit;
            isThis = 1;
        }else {
            maxbit = o2.bit;
            minbit = this->bit;
            isThis = 0;
        }
        
        int up=0;
        for(int i=0;i<minbit;++i) {
            ret.n[i] = this->n[i] + o2.n[i] + up;
            up = ret.n[i] / base;
            ret.n[i] %= base;
        }
        for(int i=minbit;i<maxbit;++i) {
            if(isThis) {
                ret.n[i] = this->n[i] + up;
                up = ret.n[i] / base;
                ret.n[i] %= base;
            }else {
                ret.n[i] = o2.n[i] + up;
                up = ret.n[i] / base;
                ret.n[i] %= base;
            }
        }
        ret.bit = maxbit;
        
        if(up) {
            ret.n[ret.bit] = up; // unsafe
            ++ret.bit;
        }
        return ret;
    }

    BigInt operator-(const BigInt&o2) {
        BigInt ret;
        int maxbit,minbit;
        if(this->bit > o2.bit) {
            maxbit = this->bit;
            minbit = o2.bit;
        }else {
            maxbit = o2.bit;
            minbit = this->bit;
        }

        if(*this>=o2) {
            ret = big_minus_small(*this,o2,maxbit,minbit);
        }else {
            ret = big_minus_small(o2,*this,maxbit,minbit);
            ret.isPositive = false;
        }

        return ret;
    }

    BigInt operator/(const BigInt&o2) {
        BigInt ret,tmp=*this;
        ret.bit = 0;
        if(*this<=o2) {
            ret.n[0] = 0;
            ret.bit = 1;
        }else {
            int bias = tmp.bit - o2.bit;
            int cnt;
            for(int i=bias;i>=0;--i) {
                for(cnt=0;!tmp.biasSmaller(o2,i);++cnt) {
                    tmp = big_minus_small(tmp,o2,tmp.bit,o2.bit,i);
                }
                ret.n[i] = cnt;
                if(!ret.bit&&cnt) ret.bit = i+1;
            }
        }
        return ret;
    }

    void print() {
        if(!isPositive) putchar('-');
        for(int i=this->bit-1;i>=0;--i) {
            putchar('0'+this->n[i]);
        }
    }

    void scan() { // only positive num
        char tmp[LENGTH];
        scanf("%s",tmp);
        this->bit = strlen(tmp);
        for(int i=0;i<this->bit;++i) {
            this->n[i] = tmp[this->bit-i-1] - '0';
        }
    }

};

int main() {
    BigInt o1,o2;
    o1.scan();
    o2.scan();
    BigInt ans = o1/o2;
    ans.print();
    
    return 0;
}```
